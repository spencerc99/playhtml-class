## Examples

### Example 1: show how many people are here

```tsx
import { usePlayContext } from '@playhtml/react';

export function UserCount() {
  const { cursors } = usePlayContext();

  return <p>People here right now: {cursors.allColors.length}</p>;
}
```

### Example 2: shared on/off state

```tsx
import { withSharedState } from '@playhtml/react';

export const ToggleSquare = withSharedState(
  {
    defaultData: { on: false },
    id: 'toggle-square',
  },
  ({ data, setData }) => (
    <button
      onClick={() => setData({ on: !data.on })}
      style={{
        width: 120,
        height: 120,
        background: data.on ? 'green' : 'red',
      }}
    />
  ),
);
```
