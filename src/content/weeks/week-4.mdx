## Examples

### Example 1: trigger a shared custom event

```tsx
import { PlayContext } from '@playhtml/react';
import { useContext, useEffect } from 'react';

const ConfettiEventType = 'confetti';

export function useConfetti() {
  const {
    dispatchPlayEvent,
    registerPlayEventListener,
    removePlayEventListener,
  } = useContext(PlayContext);

  useEffect(() => {
    const id = registerPlayEventListener(ConfettiEventType, {
      onEvent: () => {
        console.log('confetti for everyone');
      },
    });

    return () => removePlayEventListener(ConfettiEventType, id);
  }, [registerPlayEventListener, removePlayEventListener]);

  return () => dispatchPlayEvent({ type: ConfettiEventType });
}
```

### Example 2: connect it to a button

```tsx
export function CelebrationButton() {
  const triggerConfetti = useConfetti();

  return <button onClick={triggerConfetti}>Celebrate</button>;
}
```
